    public int[][] makeArray(List<Integer> list, int rows) {
        int cells = list.size() % rows != 0 ? (list.size() / rows) + 1 : list.size() / rows;
        int count = 0;
        int index = 0;
        int[][] array = new int[rows][cells];
        for (Integer in : list) {
            array[count][index++] = in;
            if (index == cells) {
                index = 0;
                count++;
            }
        }
        return array;

    }
